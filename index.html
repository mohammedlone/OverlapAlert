// Add scroll effect to header window.addEventListener('scroll', function() { const header = document.querySelector('header'); if (window.scrollY > 100) { header.style.background = 'rgba(0, 0, 0, 0.95)'; } else { header.style.background = 'rgba(255, 255, 255, 0.05)'; } }); // Clean up Mailchimp and add placeholder text function cleanMailchimp() { // Add placeholder text to email inputs const emailInputs = document.querySelectorAll('#mce-EMAIL, #mce-EMAIL-2'); emailInputs.forEach(input => { input.placeholder = 'Enter your email address'; }); // Hide unnecessary elements const elementsToRemove = [ '.brandingLogo', '.mc-referral', '.indicates-required', '.asterisk', '.helper_text', 'h2' ]; elementsToRemove.forEach(selector => { document.querySelectorAll(`.mailchimp-form ${selector}`).forEach(element => { element.style.display = 'none'; }); }); // Override Mailchimp's inline styles const forms = document.querySelectorAll('#mc_embed_signup, #mc_embed_signup_2'); forms.forEach(form => { form.style.background = 'transparent'; form.style.width = '100%'; form.style.fontFamily = 'Inter, sans-serif'; }); } // Run cleanup after delays to ensure Mailchimp form loads setTimeout(cleanMailchimp, 500); setTimeout(cleanMailchimp, 1500); // Also run on window load and DOM ready window.addEventListener('load', cleanMailchimp); document.addEventListener('DOMContentLoaded', cleanMailchimp);
